-- 1. Tabel users (manager & kasir)
CREATE TABLE users (
    id_user INT AUTO_INCREMENT PRIMARY KEY,
    nama VARCHAR(100) NOT NULL,
    username VARCHAR(50) UNIQUE NOT NULL,
    password VARCHAR(255) NOT NULL,
    email VARCHAR(100),
    role ENUM('manager', 'kasir') NOT NULL,
    status ENUM('aktif', 'nonaktif') DEFAULT 'aktif',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 2. Tabel kategori produk
CREATE TABLE kategori (
    id_kategori INT AUTO_INCREMENT PRIMARY KEY,
    nama_kategori VARCHAR(100) NOT NULL,
    icon VARCHAR(50) DEFAULT 'fa-tag'
);

-- 3. Tabel produk/menu (Revisi: Tambah Cascade pada Foreign Key)
CREATE TABLE products (
    id_product INT AUTO_INCREMENT PRIMARY KEY,
    id_kategori INT NOT NULL,
    nama_product VARCHAR(150) NOT NULL,
    harga INT NOT NULL,
    gambar VARCHAR(255) DEFAULT NULL,
    warna VARCHAR(10) DEFAULT '#4e3629',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    -- REVISI: ON DELETE CASCADE agar jika kategori dihapus, produk ikut terhapus/terupdate
    FOREIGN KEY (id_kategori) REFERENCES kategori(id_kategori) ON DELETE CASCADE 
);

-- 4. Tabel transaksi (Revisi: Tambah Debit & Kredit)
CREATE TABLE transaksi (
    id_transaksi INT AUTO_INCREMENT PRIMARY KEY,
    id_kasir INT NOT NULL,
    tanggal_transaksi DATETIME DEFAULT CURRENT_TIMESTAMP,
    total INT NOT NULL,
    uang_bayar INT NOT NULL,
    kembalian INT NOT NULL,
    -- REVISI: Tambah enum debit dan kredit
    metode_bayar ENUM('tunai', 'qris', 'debit', 'kredit') DEFAULT 'tunai', 
    status ENUM('selesai', 'batal', 'pending') DEFAULT 'selesai',
    FOREIGN KEY (id_kasir) REFERENCES users(id_user) ON DELETE RESTRICT
);

-- 5. Tabel detail transaksi (Revisi: Cascade Delete)
CREATE TABLE detail_transaksi (
    id_transaksi INT NOT NULL,
    id_product INT NOT NULL,
    nama_product VARCHAR(150) NOT NULL, 
    harga INT NOT NULL,                 
    jumlah INT NOT NULL,
    subtotal INT NOT NULL,
    -- REVISI: ON DELETE CASCADE agar jika transaksi dihapus, detailnya otomatis hilang
    FOREIGN KEY (id_transaksi) REFERENCES transaksi(id_transaksi) ON DELETE CASCADE,
    FOREIGN KEY (id_product) REFERENCES products(id_product) ON DELETE RESTRICT
);

-- 6. Tabel absensi kasir
CREATE TABLE absensi (
    id INT AUTO_INCREMENT PRIMARY KEY,
    id_user INT NOT NULL,
    tanggal DATE NOT NULL,
    jam_masuk TIME,
    jam_keluar TIME,
    total_shift INT DEFAULT 0,
    status ENUM('hadir', 'izin', 'alpha') DEFAULT 'hadir',
    FOREIGN KEY (id_user) REFERENCES users(id_user) ON DELETE CASCADE
);
